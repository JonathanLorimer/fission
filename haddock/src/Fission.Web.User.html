type API
  =   RegisterRoute
 :<|> Create.PasswordAPI
 :<|> WhoAmIRoute
 :<|> VerifyRoute
 :<|> VerifyEmailRoute
 :<|> UpdatePublicKeyRoute
 :<|> UpdateDataRoute
 :<|> ResetRoute

type Auth
  = Auth.HigherOrder

type RegisterRoute
  = Auth.RegisterDID
    :> Create.API

type WhoAmIRoute
  = "whoami"
    :> Auth
    :> WhoAmI.API

type VerifyRoute
  = "verify"
    :> Auth
    :> Verify.API

type VerifyEmailRoute
  = "email"
    :> "verify"
    :> VerifyEmail.API

type UpdatePublicKeyRoute
  = "did"
    :> Auth
    :> UpdatePublicKey.API

type UpdateDataRoute
  = "data"
    :> Auth
    :> UpdateData.API

type ResetRoute
  = "reset_password"
    :> Auth
    :> Reset.API
